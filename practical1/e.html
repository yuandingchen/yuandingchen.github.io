How you can check browser support for particular tag or attribute?
function testInputFormSupport() { var input = document.createElement('input'), form = document.createElement('form'), formId = 'test-input-form-reference-support'; // Id of the remote form form.id = formId; // Add form and input to DOM document.body.appendChild(form); document.body.appendChild(input); // Add reference of form to input input.setAttribute('form', formId); // Check if reference exists var res = !(input.form == null); // Remove elements document.body.removeChild(form); document.body.removeChild(input); return res; }
